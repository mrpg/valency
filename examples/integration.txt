#!/usr/bin/valency

set &simpson {
	-- simpson [a] [b] [iterations] [function] [&result]

	set &a #1
	set &b #2
	set &n #3
	set &f #4

	/ (- b a) n &h

	f (/ (+ a h) 2) &sum1
	set &sum2 0
	set &i 1
	set &cond (<= i (- n 1))

	while &cond {
		+ sum1 (f (+ a (* h i) (/ h 2))) &sum1
		+ sum2 (f (+ a (* h i))) &sum2
		+ i 1 &i
		set &cond (<= i (- n 1))
	}

	* (/ h 6) (+ (f a) (f b) (* 4 sum1) (* 2 sum2)) &#5
}

set &f {
	^ #1 3 &#2
}

print "I=[0; 1.8], A = " (simpson 0 1.8 3 f)
print
